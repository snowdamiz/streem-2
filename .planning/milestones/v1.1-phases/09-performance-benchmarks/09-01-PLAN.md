**apps/bench/suites/signal.mjs** — signal read+write throughput:
```js
import { Bench } from 'tinybench'
import { signal, createRoot } from '/core'
import { signal as preactSignal } from '@preact/signals-core'
import { createSignal, createRoot as solidRoot } from 'solid-js'

export async function run() {
  const bench = new Bench({ warmupIterations: 1000, iterations: 5000, name: 'signal read+write' })

  // /core
  bench.add('/core signal', () => {
    createRoot((dispose) => {
      const s = signal(0)
      s.set(1)
      const v = s()
      dispose()
      return v
    })
  })

  // Preact signals
  bench.add('@preact/signals-core signal', () => {
    const s = preactSignal(0)
    s.value = 1
    const v = s.value
    return v
  })

  // SolidJS
  bench.add('solid-js createSignal', () => {
    solidRoot((dispose) => {
      const [get, set] = createSignal(0)
      set(1)
      const v = get()
      dispose()
      return v
    })
  })

  await bench.run()
  return bench
}
```

**apps/bench/suites/effect.mjs** — effect re-run throughput:
```js
import { Bench } from 'tinybench'
import { signal, effect, createRoot } from '/core'
import { signal as preactSignal, effect as preactEffect } from '@preact/signals-core'
import { createSignal, createEffect, createRoot as solidRoot } from 'solid-js'

export async function run() {
  const bench = new Bench({ warmupIterations: 1000, iterations: 5000, name: 'effect re-run' })

  bench.add('/core effect', () => {
    createRoot((dispose) => {
      const s = signal(0)
      let runs = 0
      effect(() => { s(); runs++ })
      s.set(1)
      dispose()
      return runs
    })
  })

  bench.add('@preact/signals-core effect', () => {
    const s = preactSignal(0)
    let runs = 0
    const cleanup = preactEffect(() => { s.value; runs++ })
    s.value = 1
    cleanup()
    return runs
  })

  bench.add('solid-js createEffect', () => {
    solidRoot((dispose) => {
      const [get, set] = createSignal(0)
      let runs = 0
      createEffect(() => { get(); runs++ })
      set(1)
      dispose()
      return runs
    })
  })

  await bench.run()
  return bench
}
```

**apps/bench/run.mjs** — prints a results table for all suites:
```js
import { run as runSignal } from './suites/signal.mjs'
import { run as runComputed } from './suites/computed.mjs'
import { run as runEffect } from './suites/effect.mjs'

function printResults(bench) {
  console.log(`\n## ${bench.name}`)
  console.log('')
  const rows = bench.tasks.map(t => ({
    name: t.name,
    'ops/sec': Math.round(t.result.hz).toLocaleString(),
    'avg (ns)': (t.result.mean * 1e6).toFixed(2),
    samples: t.result.samples.length,
  }))
  // Print table
  const headers = Object.keys(rows[0])
  const widths = headers.map(h => Math.max(h.length, ...rows.map(r => String(r[h]).length)))
  const hr = widths.map(w => '-'.repeat(w + 2)).join('|')
  const fmt = row => headers.map((h, i) => (' ' + String(row[h])).padEnd(widths[i] + 2)).join('|')
  console.log('|' + headers.map((h, i) => (' ' + h).padEnd(widths[i] + 2)).join('|') + '|')
  console.log('|' + hr + '|')
  rows.forEach(r => console.log('|' + fmt(r) + '|'))
}

const node = process.version
console.log(`# /core Signal Benchmarks`)
console.log(`Node: ${node} | Date: ${new Date().toISOString().slice(0, 10)}`)

const signalBench = await runSignal()
printResults(signalBench)

const computedBench = await runComputed()
printResults(computedBench)

const effectBench = await runEffect()
printResults(effectBench)

console.log('\nDone.')
```
