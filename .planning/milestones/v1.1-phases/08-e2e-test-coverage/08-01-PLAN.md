Create apps/e2e/tests/cli-scaffold.spec.ts implementing TEST-01.

Strategy: The test does NOT use Playwright's browser — it uses Node.js `spawnSync` within the test body (Playwright tests can run Node code without a browser page). This is legitimate for CLI E2E since the artifact being tested is a terminal process, not a web page.

Implementation steps:
1. Build `create-streem` package before running (using `spawnSync('pnpm', ['build'], { cwd: createStreemPkgDir })`)
2. Create a temp directory for the scaffolded project output
3. Spawn `node dist/index.js` from `packages/create-streem/` with stdin piped:
   - Write "test-scaffold\n" (project name)
   - Write "\n" (select npm — first option, Enter)
   - Close stdin
4. After CLI exits, verify the scaffolded directory was created
5. In the scaffolded directory, run `npm install` (the CLI selected npm)
6. Run `npm run build` — assert exit code 0

```typescript
import { test, expect } from '@playwright/test'
import { spawnSync, spawn } from 'node:child_process'
import { mkdtempSync, existsSync, rmSync } from 'node:fs'
import { join, resolve } from 'node:path'
import { tmpdir } from 'node:os'
import { fileURLToPath } from 'node:url'

const MONOREPO_ROOT = resolve(fileURLToPath(import.meta.url), '../../../..')
const CREATE_STREEM_DIR = join(MONOREPO_ROOT, 'packages/create-streem')
const PROJECT_NAME = 'test-scaffold'

test.describe('TEST-01: create-streem CLI scaffold', () => {
  let tmpDir: string

  test.beforeAll(() => {
    // Build the CLI first — ensures dist/index.js is fresh
    const buildResult = spawnSync('pnpm', ['build'], {
      cwd: CREATE_STREEM_DIR,
      encoding: 'utf-8',
      shell: process.platform === 'win32',
    })
    if (buildResult.status !== 0) {
      throw new Error(`create-streem build failed:\n${buildResult.stderr}`)
    }
    // Create isolated temp directory for scaffold output
    tmpDir = mkdtempSync(join(tmpdir(), 'streem-e2e-'))
  })

  test.afterAll(() => {
    // Clean up temp directory after tests
    if (tmpDir && existsSync(tmpDir)) {
      rmSync(tmpDir, { recursive: true, force: true })
    }
  })

  test('scaffolds a project that builds successfully', async () => {
    // Run the CLI with piped stdin — answer prompts automatically
    // @clack/prompts reads via readline; piping newline-delimited answers works
    // when stdin is not a TTY (readline falls through non-interactively)
    await new Promise<void>((resolve, reject) => {
      const cli = spawn('node', ['dist/index.js'], {
        cwd: CREATE_STREEM_DIR,
        env: { ...process.env, FORCE_COLOR: '0' },
        stdio: ['pipe', 'pipe', 'pipe'],
      })

      let stdout = ''
      let stderr = ''
      cli.stdout.on('data', (d: Buffer) => { stdout += d.toString() })
      cli.stderr.on('data', (d: Buffer) => { stderr += d.toString() })

      // Feed answers: project name, then package manager selection
      // @clack/prompts text prompt: type name + Enter
      // @clack/prompts select prompt: Enter (first option = npm)
      cli.stdin.write(`${PROJECT_NAME}\n`)
      setTimeout(() => {
        cli.stdin.write('\n')  // select npm
        cli.stdin.end()
      }, 500)

      cli.on('close', (code) => {
        if (code === 0) {
          resolve()
        } else {
          reject(new Error(`CLI exited ${code}\nstdout: ${stdout}\nstderr: ${stderr}`))
        }
      })
    })

    // Verify scaffolded directory was created relative to CLI cwd
    // NOTE: create-streem resolves targetDir = resolve(process.cwd(), projectName)
    // When cwd=CREATE_STREEM_DIR, the project appears at CREATE_STREEM_DIR/test-scaffold
    // Move it to tmpDir to keep repo clean
    const scaffoldedDir = join(CREATE_STREEM_DIR, PROJECT_NAME)
    expect(existsSync(scaffoldedDir)).toBe(true)
    expect(existsSync(join(scaffoldedDir, 'package.json'))).toBe(true)
    expect(existsSync(join(scaffoldedDir, 'src/main.tsx'))).toBe(true)

    // Install dependencies (CLI selected npm and already ran install, but verify ourselves)
    const installResult = spawnSync('npm', ['install'], {
      cwd: scaffoldedDir,
      encoding: 'utf-8',
      timeout: 90_000,
      shell: process.platform === 'win32',
    })
    expect(installResult.status, `npm install failed:\n${installResult.stderr}`).toBe(0)

    // Run pnpm build — this is the core assertion: the scaffold builds without errors
    const buildResult = spawnSync('npm', ['run', 'build'], {
      cwd: scaffoldedDir,
      encoding: 'utf-8',
      timeout: 60_000,
      shell: process.platform === 'win32',
    })
    expect(buildResult.status, `npm run build failed:\n${buildResult.stderr}\n${buildResult.stdout}`).toBe(0)

    // Move to tmpDir for cleanup
    const { renameSync } = await import('node:fs')
    renameSync(scaffoldedDir, join(tmpDir, PROJECT_NAME))
  })
})
```

Key implementation notes:
- `spawnSync` for synchronous CLI steps (build, install, build-check)
- `spawn` + stdin write for interactive CLI — pipe in project name then Enter for package manager
- 500ms delay between answers to allow @clack/prompts to process first prompt
- The CLI runs with cwd=CREATE_STREEM_DIR so scaffolded dir appears there; move to tmpDir after
- `FORCE_COLOR: '0'` disables ANSI codes in stdout for cleaner error messages
- `timeout: 90_000` for npm install (network-dependent)
