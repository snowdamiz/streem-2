Create `apps/landing/vite.config.ts` (MPA with two entry points, streemHMR, Shoelace asset copy):
```typescript
import { defineConfig } from 'vite'
import { streemHMR } from 'streem'
import { viteStaticCopy } from 'vite-plugin-static-copy'
import { resolve } from 'path'

// Change to '/' if a custom domain is configured for this GitHub Pages site
const BASE_URL = process.env.VITE_BASE_URL ?? '/streem-2/'

export default defineConfig({
  base: BASE_URL,
  plugins: [
    streemHMR(),
    viteStaticCopy({
      targets: [{
        src: 'node_modules/@shoelace-style/shoelace/dist/assets/**/*',
        dest: 'shoelace_assets'
      }]
    })
  ],
  build: {
    rollupOptions: {
      input: {
        main: resolve(__dirname, 'index.html'),
        docs: resolve(__dirname, 'docs/index.html'),
      }
    }
  }
})
```

Create `apps/landing/src/main.tsx` — CRITICAL: setBasePath MUST be called before any Shoelace component import. Do NOT import Shoelace components in this file — that happens in InstallCta.tsx. This file only sets the path so it's registered before lazy component registration:
```tsx
// MUST be first import — sets Shoelace base path before any component self-registers
import { setBasePath } from '@shoelace-style/shoelace/dist/utilities/base-path.js'
setBasePath('./shoelace_assets')

import '@shoelace-style/shoelace/dist/themes/dark.css'
import './styles/global.css'
import { render } from 'streem'
import { App } from './App'

render(App, document.getElementById('app')!)
```
