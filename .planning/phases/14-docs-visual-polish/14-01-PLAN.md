**Change 2 — Syntax-highlighted code blocks:**
Add import at top of DocsApp.tsx:
```tsx
import { highlight } from './lib/highlight'
```

Replace the `Code` component. Because `highlight()` returns an HTML string with `<span>` tags, it must be set via `innerHTML`. Streem's JSX doesn't support `dangerouslySetInnerHTML`, but we can use a ref pattern via `onMount`:

```tsx
import { onMount } from 'streem'
import { highlight } from './lib/highlight'

function Code({ children }: { children: string }) {
  let codeEl: HTMLElement
  onMount(() => {
    codeEl.innerHTML = highlight(children)
  })
  return (
    <pre class="doc-pre">
      <code ref={(el: HTMLElement) => { codeEl = el }}></code>
    </pre>
  ) as unknown as Node
}
```

Wait — Streem does not use a `ref` prop API. Instead, assign innerHTML directly using a JS element created in JSX. The simplest safe approach: create the `<pre>` element imperatively and set innerHTML on the `<code>` child:

```tsx
function Code({ children }: { children: string }) {
  const pre = document.createElement('pre')
  pre.className = 'doc-pre'
  const code = document.createElement('code')
  code.innerHTML = highlight(children)
  pre.appendChild(code)
  return pre
}
```

This is safe here because `children` is always a string literal passed from DocsApp (no user input), and `highlight()` only wraps tokens in `<span style="color:...">` tags — no script injection surface.
