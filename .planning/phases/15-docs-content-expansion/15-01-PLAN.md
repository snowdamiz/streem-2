- A paragraph: "Batch multiple signal writes so downstream effects run only once:"
- A Code block:
```ts
import { signal, effect, batch } from 'streem'

const x = signal(0)
const y = signal(0)

effect(() => console.log(x(), y()))

// Without batch: effect fires twice
x.set(1)
y.set(1)

// With batch: effect fires once
batch(() => {
  x.set(2)
  y.set(2)
})
```
- A paragraph: "TypeScript tip: use `signal<string | null>(null)` for optional values. The computed type is inferred automatically."

  return (

      // --- FORMS ---
      <h3 class="doc-section-subtitle">Form handling</h3>
      <p>Bind each field to a signal and derive a computed for the submit payload:</p>
      <Code>{`import { signal, computed } from 'streem'

function LoginForm() {
  const email = signal('')
  const password = signal('')
  const isValid = computed(() => email().includes('@') && password().length >= 8)

  function handleSubmit(e: Event) {
    e.preventDefault()
    if (!isValid()) return
    fetch('/api/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email: email(), password: password() }),
    })
  }

  return (
    <form on:submit={handleSubmit}>
      <input
        type="email"
        on:input={(e) => email.set((e.target as HTMLInputElement).value)}
        placeholder="Email"
      />
      <input
        type="password"
        on:input={(e) => password.set((e.target as HTMLInputElement).value)}
        placeholder="Password"
      />
      <button type="submit" disabled={() => !isValid()}>Log in</button>
    </form>
  )
}`}</Code>

      // --- DATA FETCHING ---
      <h3 class="doc-section-subtitle">Data fetching</h3>
      <p>Use a signal for the data and a status signal for loading/error states:</p>
      <Code>{`import { signal, onMount } from 'streem'

interface User { id: number; name: string }

function UserProfile({ userId }: { userId: number }) {
  const user = signal<User | null>(null)
  const status = signal<'loading' | 'done' | 'error'>('loading')

  onMount(async () => {
    try {
      const res = await fetch(\`/api/users/\${userId}\`)
      user.set(await res.json())
      status.set('done')
    } catch {
      status.set('error')
    }
  })

  return (
    <div>
      <Show when={() => status() === 'loading'}>
        {() => <p>Loading...</p>}
      </Show>
      <Show when={() => status() === 'done' && user() !== null}>
        {() => <p>Hello, {() => user()!.name}</p>}
      </Show>
      <Show when={() => status() === 'error'}>
        {() => <p>Failed to load user.</p>}
      </Show>
    </div>
  )
}`}</Code>

      // --- SHARED STATE ---
      <h3 class="doc-section-subtitle">Shared state</h3>
      <p>Export signals from a module — any component that imports them shares the same reactive state:</p>
      <Code>{`// store/auth.ts
import { signal, computed } from 'streem'

export const currentUser = signal<{ name: string; role: string } | null>(null)
export const isLoggedIn = computed(() => currentUser() !== null)
export const isAdmin = computed(() => currentUser()?.role === 'admin')

export function login(user: { name: string; role: string }) {
  currentUser.set(user)
}

export function logout() {
  currentUser.set(null)
}

// components/Header.tsx
import { currentUser, isLoggedIn, logout } from '../store/auth'

function Header() {
  return (
    <header>
      <Show when={isLoggedIn}>
        {() => (
          <div>
            <span>Hello, {() => currentUser()!.name}</span>
            <button on:click={logout}>Log out</button>
          </div>
        )}
      </Show>
    </header>
  )
}`}</Code>

interface TradeEvent { symbol: string; price: number; volume: number }

function LiveTradesFeed() {
  const [rawTrade] = fromWebSocket<TradeEvent>('wss://trades.example.com')
  const displayTrade = throttle(rawTrade, 100) // max 10 updates/sec

  const trades = signal<TradeEvent[]>([])

  effect(() => {
    const t = displayTrade()
    if (t == null) return
    trades.set([t, ...trades().slice(0, 49)]) // keep last 50
  })

  return (
    <ul>
      <For each={trades} by={(t, i) => i}>
        {(trade) => (
          <li>{() => \`\${trade().symbol} @ \${trade().price}\`}</li>
        )}
      </For>
    </ul>
  )
}`}</Code>
